<h2>解析：</h2>
绘制由圆弧和叶子的组合。
<h2>答案：</h2>
<pre class="prettyprint linenums">import turtle as t
def yuanhu(r, n):
    # 用 循环绘制需要增加左转方法
    # for i in range(0,n):
    #     t.fd(0.0174*r)
    #     t.rt(1)
    #     t.lt(1)
    # 用 circle 方法绘制
    t.circle(-r, n)
    return
 
def yezi(r, n):
    for i in range(0,2):
        yuanhu(r,n)
        t.rt(180-n)
    return

n=3
t.lt(90)
for i in range(0, n):
    yuanhu(50, 150/n)
    t.lt(90)
    yezi(10, 90)
    t.rt(90)
    yezi(10, 90)

for i in range(0, n):
    yuanhu(-50, 150/n)
    t.lt(90)
    yezi(10, 90)
    t.rt(90)
    yezi(10, 90)
</pre>
