<h2>答案：</h2>
<pre class="prettyprint linenums">import turtle as t
def yuanhu(r, n):
    # 用 循环绘制
    # for i in range(0,n):
    #     t.fd(0.0174*r)
    #     t.rt(1)
    # 用 circle 方法绘制
    t.circle(-r, n)
    return
 
def yezi(r, n):
    for i in range(0,2):
        yuanhu(r,n)
        t.rt(180-n)
    return

for i in range(0, 4):
    s=20
    for j in range(0, 3):
        yezi(s, 90)
        s=s+10
    t.rt(90)
