<h2>解析：</h2>
绘制由圆弧和叶子组成的花环。
<h2>答案：</h2>
<pre class="prettyprint linenums">import turtle as t
def yuanhu(r, n):
    # 用 循环绘制
    # for i in range(0,n):
    #     t.fd(0.0174*r)
    #     t.rt(1)
    # 用 circle 方法绘制
    t.circle(-r, n)
 
def yezi(r, n):
    for i in range(0,2):
        yuanhu(r,n)
        t.rt(180-n)
# 3 对叶子
n=3
# 4 对叶子
# n=4
# 5 对叶子
# n=5
for i in range(0, n):
    yuanhu(50, 360/n)
    t.lt(90)
    yezi(10, 90)
    t.rt(90)
    yezi(10, 90)
