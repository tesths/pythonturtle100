<pre class="prettyprint linenums">import turtle as t
import math
def yuanhu(r, n):
    # 用 循环绘制
    # for i in range(0,n):
    #     t.fd(0.0174*r)
    #     t.rt(1)
    # 用 circle 方法绘制
    t.circle(-r, n)

r=60
n=6
m=100

for i in range(0, n):
    # 计算使用
    # print(r*math.sin(math.radians(m/2))*2)
    t.fd(91.9)
    t.rt(360/n)

t.lt(90-(180-m)/2)
for i in range(0, n):
    yuanhu(r, m)
    t.lt(m)
    t.rt(360/n)
