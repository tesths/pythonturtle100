方法一
<pre class="prettyprint linenums">import turtle as t
def banyuan(r):
    for i in range(0, 180):
        t.fd(0.0174*r)
        t.rt(1)
    return

# 图案一
n=3
# 图案二
# n=4
# 图案三
# n= 5
t.lt(90)
for i in range(0, n):
    banyuan(50)
    t.rt((n-2)*180/n)
